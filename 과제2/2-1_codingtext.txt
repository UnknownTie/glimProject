#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>   

int solution(int n, int k) {
    // 예외 처리 
    if (0 >= n || n >= 1000 || (n / 10) > k || k >= 1000)
        return 0;

    int rtn = 0; 
    // 양꼬치(n) : 12000 , 음료수(k) : 2000
    static int payFood = 12000;
    static int payDrink = 2000;
    //계산
    int bill = (n * payFood) + ((k- (n/10)) * payDrink) ;
    rtn = bill;
    
    return rtn;
}
